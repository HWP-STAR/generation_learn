dl_generation_learn/          # 项目根目录
├── README.md                 # 项目说明：学习目标、环境配置、运行方式
├── requirements.txt          # 依赖清单：torch、torchvision、numpy等
├── .gitignore                # 忽略不需要版本控制的文件（如缓存、日志）
├── configs/                  # 配置文件目录（统一管理超参数，避免硬编码）
│   ├── base.yaml             # 基础配置：数据集路径、训练设备、通用超参数
│   ├── gan_mnist.yaml        # 针对MNIST的GAN配置
│   └── diffusion_cifar10.yaml # 针对CIFAR10的扩散模型配置
├── data/                     # 数据集目录（自动下载/手动存放）
│   ├── raw/                  # 原始数据集（如MNIST、CIFAR10的原始文件）
│   ├── processed/            # 预处理后的数据集（可选，比如归一化、裁剪后的文件）
│   └── custom/               # 自定义数据集（比如自己收集的图片）
├── models/                   # 模型定义目录（按模型类型拆分，便于对比学习）
│   ├── __init__.py           # 让目录成为Python模块，方便导入
│   ├── base_model.py         # 基础模型类（封装训练/验证通用逻辑）
│   ├── gan/                  # GAN相关模型
│   │   ├── __init__.py
│   │   ├── dcgan.py          # DCGAN实现
│   │   └── wgan_gp.py        # WGAN-GP实现
│   ├── vae/                  # VAE相关模型
│   │   ├── __init__.py
│   │   └── vanilla_vae.py    # 基础VAE实现
│   └── diffusion/            # 扩散模型相关
│       ├── __init__.py
│       └── ddpm.py           # DDPM基础实现
├── utils/                    # 工具函数目录（复用性代码，避免重复编写）
│   ├── __init__.py
│   ├── data_utils.py         # 数据加载、预处理函数
│   ├── train_utils.py        # 训练循环、梯度裁剪、早停等函数
│   ├── vis_utils.py          # 结果可视化（生成图片展示、损失曲线绘制）
│   └── logger_utils.py       # 日志记录、实验结果保存
├── experiments/              # 实验运行目录（存放训练脚本和实验结果）
│   ├── train_gan_mnist.py    # 训练MNIST-GAN的脚本
│   ├── train_vae_cifar10.py  # 训练CIFAR10-VAE的脚本
│   └── results/              # 实验结果存放
│       ├── gan_mnist/        # 某一次GAN实验的结果
│       │   ├── checkpoints/  # 模型权重 checkpoint
│       │   ├── logs/         # 训练日志（损失、精度等）
│       │   └── samples/      # 生成的样本图片
│       └── diffusion_cifar10/
├── notebooks/                # Jupyter Notebook（学习调试专用）
│   ├── 01_mnist_gan_demo.ipynb  # 入门：MNIST-GAN快速演示
│   ├── 02_vae_principle.ipynb   # 原理学习：VAE数学推导+代码
│   └── 03_diffusion_visualize.ipynb # 扩散模型过程可视化
└── docs/                     # 学习笔记/文档（加深理解）
    ├── gan_principle.md      # GAN原理笔记
    ├── vae_formula.md        # VAE公式推导
    └── diffusion_notes.md    # 扩散模型学习总结
